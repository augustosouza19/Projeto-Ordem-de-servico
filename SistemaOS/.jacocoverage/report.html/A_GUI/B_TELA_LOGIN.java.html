    private void btnLogActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btnLogActionPerformed

        // MÉTODO LOGAR
        String sql;
        String perfil;

        // QUERY PARA SELECIONAR TUDO DA TABELA USUÁRIO COM O LOGIN E SENHA DO BANCO DE DADOS
        sql = " select * from tbusuario where login = ? and senha = ? ";

        try 
        {
            // PREPARAR CONSULTA AO BANCO DE DADOS EM FUNÇÃO DO QUE FOI DIGITADO NA CAIXA DE TEXTO
            // O "?" É SUBSTITUÍDO PELO CONTEÚDO DAS VARIÁVEIS
            pst = conexao.prepareStatement( sql );
            
            // PEGAR LOGIN E SENHA DO USUÁRIO DO BANCO DE DADOS E COLOCAR NOS CAMPOS DE TEXTO
            pst.setString( 1, txtUsu.getText() );
            pst.setString( 2, txtSen.getText() );

            // EXECUTA A QUERY
            rs = pst.executeQuery();

            // SE EXISTIR USUÁRIO E SENHA CORRESPONDENTE
            if ( rs.next() ) 
            {
                // OBTÉM O CONTEÚDO PERFIL DA TABELA TBUSUARIO; DATABASE = DBINFOX
                perfil = rs.getString(6);
                System.out.println( perfil );
                
                // TRATAMENTO DO PERFIL DO USUÁRIO
                if( perfil.equals( "admin" ) )
                {
                    // TELA DE LOGIN AUTOMATICAMENTE FECHA JUNTO COM A CONEXÃO
                    this.dispose();

                    // TELA PRINCIPAL ABRE
                    C_TELA_PRINCIPAL principal = new C_TELA_PRINCIPAL(); 
                    principal.setVisible( true );
                    
                    // MENREL, MENCADUSU, LBLUSU, LBLUSU SÃO VARIÁVEIS PÚBLICAS E ESTÁTICAS
                    // LIBERANDO CAMPOS MERNREL E MENCADUSU
                    C_TELA_PRINCIPAL.MenRel.setEnabled( true );
                    C_TELA_PRINCIPAL.MenCadUsu.setEnabled( true );
                    
                    // MUDANDO A COR DA LUSU PARA VERMELHO
                    C_TELA_PRINCIPAL.lblUsu.setText( rs.getString( 2 ) );
                    C_TELA_PRINCIPAL.lblUsu.setForeground( Color.RED );
                }
                
                else
                {
                    // TELA DE LOGIN AUTOMATICAMENTE FECHA JUNTO COM A CONEXÃO
                    this.dispose();
                    
                    // TELA PRINCIPAL ABRE
                    C_TELA_PRINCIPAL principal = new C_TELA_PRINCIPAL();  
                    
                    principal.setVisible( true );
                    
                    // LBLUSU, LBLUSU SÃO VARIÁVEIS PÚBLICAS E ESTÁTICAS
                    // MUDANDO A COR DA LUSU PARA AZUL
                    C_TELA_PRINCIPAL.lblUsu.setText( rs.getString( 2 ) );
                    C_TELA_PRINCIPAL.lblUsu.setForeground( Color.BLUE );
                }
            }
            
            else
            {
                JOptionPane.showMessageDialog( null, " ACESSO NEGADO ", " ERRO DE ACESSO ", JOptionPane.ERROR_MESSAGE );
                cont ++;
            }
            
            if ( cont > 3 )
            {
                JOptionPane.showMessageDialog( null," ACESSO BLOQUEADO ", " ERRO DE ACESSO ", JOptionPane.ERROR_MESSAGE );
                System.exit(0);
            }
        }
        
        catch ( SQLException | HeadlessException e ) 
        {
            // APOIO AO ERRO
            System.out.println( e );
        }
    }//GEN-LAST:event_btnLogActionPerformed
